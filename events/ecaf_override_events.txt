namespace = unity_patrons
planet_event = {
	id = unity_patrons.1
	hide_window = yes
	is_triggered_only = yes
	location = root
	pre_triggers = {
		has_owner = yes
	}
	trigger = {
		exists = owner
		owner = {
			has_valid_civic = civic_peacekeepers
		}
	}
	immediate = {
		owner = {
			add_modifier = {
				modifier = "unity_patrons"
				days = 3650
			}
		}
	}
}

namespace = glorious_fleet

# from = owner of fleet 2 (destroyed)
# fromfrom = fleet 1
# fromfromfrom = fleet 2
country_event = {
	id = glorious_fleet.0
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_valid_civic = civic_naval_traditions
		NOR = {
			has_country_flag = glorious_fleet
			has_country_flag = unglorious_fleet
		}
	}
	immediate = {
		add_modifier = {
			modifier = "glorious_fleet"
			days = 30
		}
		set_timed_country_flag = {
			flag = glorious_fleet
			days = 30
		}
	}
}

namespace = anglers_fix
## Anglers Fix
country_event = {
	id = anglers_fix.1000
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_ai = yes
		has_civic = civic_anglers
		capital_scope = { is_wet = no }
	}
	immediate = {
		force_remove_civic = civic_anglers
		change_government = {
			civics = random
			cooldown = no
		}
		species = { remove_trait = trait_aquatic }
	}
}
