building_commercial_zone = {
	base_buildtime = @b1_time
	exempt_from_ai_planet_specialization = no
	category = trade

	building_sets = {
		urban
		trade
	}

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOR = {
			has_modifier = slave_colony
			has_modifier = penal_colony
		}
		NOT = { has_building = building_low_tech_capital }
		OR = {
			has_planet_flag = ignore_ai_building_limitations
			owner = { is_ai = no }
			# AI is allowed to build one of these unless they're a megacorp or this is a trade planet
			owner = { is_megacorp = yes }
			has_any_trade_zone = yes
			NOR = {
				has_building = building_commercial_zone
				has_building = building_commercial_megaplex
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			has_modifier = penal_colony
		}
	}

	allow = {
		is_occupied_flag = no # Ariphaos
	}

	convert_to = {
		building_maintenance_depot
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	planet_modifier = {
		job_clerk_add = 1
		job_trader_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { NOT = { has_valid_civic = civic_consumerism } }
		}
		modifier = {
			job_clerk_add = 100
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_consumerism }
		}
		modifier = {
			job_trader_add = 100
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_mercantile_commercial_enterprise
			}
		}
		modifier = {
			job_trader_add = 100
		}
	}

	inline_script = {
		script = jobs/trader_add
		AMOUNT = @building_static_jobs
	}

	triggered_desc = {
		potential = {
			exists = owner
			owner = { NOT = { has_valid_civic = civic_consumerism } }
		}
		text = job_clerk_effect_desc
	}

	prerequisites = {
		"tech_interplanetary_commerce"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}
}

building_galactic_stock_exchange = {
	base_buildtime = @b2_time
	base_cap_amount = 1

	category = trade

	potential = {
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}

	allow = {
		has_major_upgraded_capital = yes
		is_occupied_flag = no # Ariphaos
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_galactic_stock_exchange
						is_starbase_building_building = ring_galactic_stock_exchange
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			has_modifier = resort_colony
			has_modifier = penal_colony
		}
	}

	planet_modifier = {
		pop_trader_bonus_workforce_mult = 0.15
	}

	country_modifier = {
		country_trade_fee = -0.05
	}

	inline_script = {
		script = jobs/trader_add
		AMOUNT = @building_static_jobs
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			rare_crystals = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			rare_crystals = @b2_rare_upkeep
		}
	}

	triggered_planet_modifier = { ## ECAF
		potential = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_merchant_guilds
					has_valid_civic = civic_state_monopoly
				}
			}
		}
		modifier = {
			job_merchant_add = 50
		}
	}

	prerequisites = {
		"tech_galactic_markets"
	}

	triggered_desc = {
		text = job_mogul_effect_desc
	}

	ai_resource_production = {
		trade = 10
	}
}
