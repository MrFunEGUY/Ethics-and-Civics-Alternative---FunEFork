building_military_academy = {
	base_buildtime = @b2_time
	empire_limit = 1
	can_be_ruined = no
	category = army

	building_sets = {
		fortress
	}

	potential = {
		exists = owner
		owner = { NOT = { has_valid_civic = civic_reanimated_armies } }
		OR = {
			owner = { is_stratocracy = yes }
			NOR = {
				has_modifier = resort_colony
				has_modifier = penal_colony
			}
		}
	}

	allow = {
		OR = {
			has_major_upgraded_capital = yes
			AND = {
				exists = owner
				owner = {
					has_valid_civic = civic_machine_tactical_algorithms
				}
			}
		}
		is_occupied_flag = no # Ariphaos
	}

	## ECAF

	country_modifier = {
		country_resource_max_ecaf_crew_add = @major_capital_crew_max_add
		country_commander_cap_add = 1
		country_naval_cap_mult = 0.02
	}

	planet_modifier = {
		planet_pops_ecaf_crew_produces_mult = 0.05
	}

	army_modifier = { # ECAF
		army_starting_experience_add = 500
	}

	triggered_country_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_machine_tactical_algorithms }

		}
		ship_evasion_mult = 0.02
		ship_tracking_mult = 0.01
		ship_shield_penetration_add = 0.05
		ship_armor_penetration_add = 0.05
		mult = owner.value:tactical_algorithms_scaling_mult
	}
	triggered_country_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_machine_tactical_algorithms }

		}
		ship_shield_hardening_add = 0.1
		mult = owner.value:tactical_algorithms_enclave_scaling_mult
	}

	triggered_country_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
			}
		}
		pop_soldier_bonus_workforce_mult = 0.15
	}

	triggered_country_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = yes
			}
		}
		pop_warrior_drone_bonus_workforce_mult = 0.15
	}

	triggered_country_modifier = {
		potential = {
			exists = owner
			owner = {
				has_origin = origin_toxic_knights
			}
		}
		pop_knight_bonus_workforce_mult = 0.05
		pop_knight_commander_bonus_workforce_mult = 0.025
	}

	triggered_planet_modifier = { # ECAF
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
				has_valid_civic = civic_martial_brotherhood
			}
		}
		modifier = {
			planet_housing_add = 200
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
		}
		cost = {
			trigger = {
				exists = owner
				owner = {
					NOT = { has_valid_civic = civic_machine_tactical_algorithms }
				}
			}
			volatile_motes = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					NOT = { has_valid_civic = civic_machine_tactical_algorithms }
				}
			}
			volatile_motes = @b2_rare_upkeep
		}
		## ECAF
		produces = {
			trigger = {
				owner = { has_valid_civic = civic_martial_brotherhood }
			}
			unity = 5
		}
		produces = {
			trigger = { owner = { has_valid_civic = civic_citizen_service } }
			ecaf_crew = @mil_academy_civic_crew_production
		}
		produces = {
			ecaf_crew = @mil_academy_crew_production
		}
	}

	ai_resource_production = {
		ecaf_crew = @crew_ai_resource_mil_academy_production
		trigger = {
			always = yes
		}
	}

	ai_resource_production = {
		ecaf_crew = @soldier_job_crew_production
		trigger = {
			OR = {
				owner = { has_valid_civic = civic_stratocracy }
				owner = { has_valid_civic = civic_citizen_service }
				owner = { has_valid_civic = civic_martial_brotherhood }
			}
		}
	}

	prerequisites = {
		"tech_centralized_command"
	}

	inline_script = {
		script = jobs/soldiers_add
		AMOUNT = @building_static_jobs
	}
}

building_stronghold = {
	base_buildtime = 240
	position_priority = 100
	can_be_ruined = no
	exempt_from_ai_planet_specialization = no

	category = army

	building_sets = {
		government
		fortress
	}

	potential = {
		exists = owner
		OR = {
			owner = { is_stratocracy = yes }
			NOT = { has_modifier = resort_colony }
		}
		owner = { is_wilderness_empire = no } # see 'building_planetary_carapace'
		OR = {
			has_planet_flag = ignore_ai_building_limitations
			owner = {
				is_ai = no
			}

			# AI is allowed to build up to three of these unless they have a fortress zone. (This can result in three under certain upgrade sequences.)
			has_any_fortress_zone = yes
			AND = {
				num_buildings = { type = building_stronghold value < 2 }
				num_buildings = { type = building_fortress value < 2 }
			}
		}
	}

	allow = {
		is_occupied_flag = no # Ariphaos
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
				NOT = { has_valid_civic = civic_private_militaries }
			}
		}
		modifier = {
			job_soldier_add = @b1_jobs
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = @b1_jobs
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
				has_valid_civic = civic_private_militaries
			}
		}
		modifier = {
			job_mercenary_add = 200
		}
	}

	resources = {
		category = planet_buildings_strongholds
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 1
		}
		## ECAF
		produces = {
			trigger = {
				exists = owner
				owner = { has_active_tradition = tr_unyielding_resistance_is_frugal }
			}
			unity = 3
		}
		produces = {
			ecaf_crew = @base_fortress_building_crew_production
		}
		produces = {
			trigger = {
				owner = { has_valid_civic = civic_citizen_service }
			}
			ecaf_crew = @base_fortress_building_civic_crew_production
		}
	}

	country_modifier = { # ECAF
		country_resource_max_ecaf_crew_add = @fortress_building_crew_max_add
	}

	prerequisites = {
		"tech_planetary_defenses"
	}

	show_tech_unlock_if = {
		is_wilderness_empire = no
	}

	inline_script = {
		script = jobs/soldiers_add
		AMOUNT = @building_static_jobs
	}

	upgrades = {
		"building_fortress"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_valid_civic = civic_private_militaries }
		}
		text = job_mercenary_effect_desc
	}

	ai_resource_production = {
		ecaf_crew = @crew_ai_resource_fortress_production
		trigger = { always = yes }
	}

	ai_resource_production = {
		ecaf_crew = @crew_ai_resource_fortress_civic_production
		trigger = { owner = { has_valid_civic = civic_citizen_service } }
	}
}
