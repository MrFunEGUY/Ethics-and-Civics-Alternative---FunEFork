building_colliseum = {
	base_buildtime = 480
	base_cap_amount = 1
	category = amenity
	potential = {
		exists = owner
		owner = { has_valid_civic = civic_slaving_despots }
	}

	allow = {
		has_upgraded_capital = yes
		has_building = building_slave_processing
		is_occupied_flag = no # Ariphaos
		owner = { is_pacifist = no }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			not = {
				has_building = building_slave_processing
			}
			owner = {
				not = {
					has_valid_civic = civic_slaving_despots
				}
			}
		}
	}

	planet_modifier = {
		job_gladiator_add = 200
		country_trade_produces_mult = 0.05
		planet_housing_add = 100
	}

	country_modifier = { # ECAF
		country_resource_max_ecaf_crew_add = @fortress_building_crew_max_add
	}

	army_modifier = {
		army_damage_mult = 0.1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_warrior_culture
			}
		}
		modifier = {
			job_gladiator_add = 100
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
			alloys = 50
		}
		upkeep = {
			energy = 2
		}
	}

	triggered_desc = {
		text = job_gladiator_effect_desc
	}

	ai_weight = {
		weight = 5
	}
	
	on_built = {
		planet = {
			owner = {
				set_timed_country_flag = {
					flag = buildings_new_alloys_building
					days = 480
				}
			}
		}
	}
}
